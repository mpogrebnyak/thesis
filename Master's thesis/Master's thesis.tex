\ddot{x}(t) = d (\dot{x}(t-\tau)-\dot{x}(t) - \lambda).
\end{equation*}

\begin{equation*}
\begin{split}
&\ddot{x}_1(t) = d (\dot{x}_0(t-\tau)-\dot{x}_1(t) - \lambda), \\ 
&\ddot{x}_2(t) = d (\dot{x}_1(t-\tau)-\dot{x}_2(t) - \lambda), \\
&\ldots \\
&\ddot{x}_N(t) = d (\dot{x}_{N-1}(t-\tau)-\dot{x}_N(t) - \lambda).
\end{split}
\end{equation*}

\begin{equation*}
